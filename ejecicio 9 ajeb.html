//un maestro desea saber que porcentaje de hombres y que 
//porcentaje de mujeres hay en un grupo de estudiantes.
Proceso ejercicio
    definir num_hombres, num_mujeres como enteros;
    definir total_estudiantes como enteros;
    definir porcentajeH, porcentajeM como enteros;
    escribir "digite el numero de hombre:";
    leer num_hombres;
    escribir "digite el numero de mujeres:";
    leer num_mujeres;
    total_estudiantes ← num_hombres+num_mujeres;
    porcentajeH ← num_hombres / total_estudiantes * 100;
    porcentajeM ← num_mujeres / total_estudiantes * 100;
    escribir "el porcentaje de hombres es:",porcentajeH,"%";
    escribir "el porcentaje de mujeres es:",porcentajeM,"%";
FinProceso
